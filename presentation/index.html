orchestrator/graph.py

graph = StateGraph(AgentState)
graph.add_node("guard_in", input_guardrails)
graph.add_node("router", router_agent)
graph.add_node("rbac", rbac_filter)
graph.add_node("retrieve", hybrid_retriever)
graph.add_node("generate", llm_generate)
graph.add_node("guard_out", output_guardrails)

graph.set_entry_point("guard_in")
graph.add_edge("guard_in", "router")
graph.add_conditional_edges(
    "router", choose_strategy,
    {"graph": "rbac", "vector": "retrieve"}
)
graph.add_edge("rbac", "retrieve")
graph.add_edge("retrieve", "generate")
graph.add_conditional_edges(
    "generate", check_confidence,
    {"low": "router", "ok": "guard_out"}
)
graph.add_edge("guard_out", END)

08 · SECURITY

RBAC на узлах графа + двусторонние Guardrails

NODE-LEVEL RBAC

Доступ ограничен до выполнения LLM

Каждый узел Neo4j имеет свойство roles: [str]

Cypher-запрос фильтрует подграф ролью пользователя

LLM никогда не видит запрещённые сущности

MATCH (d:Document)-[:CONTAINS]->(n:Norm)
WHERE $user_role IN d.roles
  AND $user_role IN n.roles
RETURN n
